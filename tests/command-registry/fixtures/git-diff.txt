+import type { AuthConfig } from './types.js';
+import { verifyToken } from './token.js';
+
+export interface ValidationResult {
+  valid: boolean;
+  sub?: string;
+  error?: string;
+}
+
+export async function validateRequest(
+  authHeader: string | null,
+  config: AuthConfig,
+): Promise<ValidationResult> {
+  if (!authHeader) {
+    return { valid: false, error: 'Missing Authorization header' };
+  }
+
+  const [scheme, token] = authHeader.split(' ');
+  if (scheme !== 'Bearer' || !token) {
+    return { valid: false, error: 'Invalid Authorization scheme' };
+  }
+
+  try {
+    const payload = await verifyToken(token, config);
+    return { valid: true, sub: payload.sub };
+  } catch {
+    return { valid: false, error: 'Invalid or expired token' };
+  }
+}

+import type { AuthConfig } from './types.js';
+import { createToken, verifyToken } from './token.js';
+
+export interface Session {
+  sub: string;
+  token: string;
+  expiresAt: Date;
+}
+
+const sessions = new Map<string, Session>();
+
+export async function createSession(
+  sub: string,
+  config: AuthConfig,
+): Promise<Session> {
+  const token = await createToken(sub, config);
+  const expiresAt = new Date(Date.now() + config.expirySeconds * 1000);
+  const session: Session = { sub, token, expiresAt };
+  sessions.set(token, session);
+  return session;
+}
+
+export function getSession(token: string): Session | undefined {
+  return sessions.get(token);
+}
+
+export function destroySession(token: string): boolean {
+  return sessions.delete(token);
+}
+
+export async function refreshSession(
+  token: string,
+  config: AuthConfig,
+): Promise<Session | undefined> {
+  const existing = sessions.get(token);
+  if (!existing) return undefined;
+  sessions.delete(token);
+  return createSession(existing.sub, config);
+}

+import type { TokenPayload } from './types.js';
+
+interface CacheEntry {
+  payload: TokenPayload;
+  cachedAt: number;
+  ttl: number;
+}
+
+export class TokenCache {
+  private readonly cache = new Map<string, CacheEntry>();
+  private readonly defaultTtl: number;
+
+  constructor(defaultTtl = 300_000) {
+    this.defaultTtl = defaultTtl;
+  }
+
+  set(token: string, payload: TokenPayload, ttl?: number): void {
+    this.cache.set(token, {
+      payload,
+      cachedAt: Date.now(),
+      ttl: ttl ?? this.defaultTtl,
+    });
+  }
+
+  get(token: string): TokenPayload | undefined {
+    const entry = this.cache.get(token);
+    if (!entry) return undefined;
+    if (Date.now() - entry.cachedAt > entry.ttl) {
+      this.cache.delete(token);
+      return undefined;
+    }
+    return entry.payload;
+  }
+
+  delete(token: string): boolean {
+    return this.cache.delete(token);
+  }
+
+  clear(): void {
+    this.cache.clear();
+  }
+
+  get size(): number {
+    return this.cache.size;
+  }
+}
